递归来枚举子集
<!-- more -->
```Go
func subsets(nums []int) [][]int {
    result := [][]int{}
    curr := []int{}

    var dfs func(int)
    dfs = func(i int){
        if i >= len(nums) {
	        // 合并到答案的时候新开变量
            result = append(result, append([]int{}, curr...))
            return
        }

        curr = append(curr, nums[i])
        dfs(i + 1)
        curr = curr[:len(curr)-1]
        dfs(i + 1)
    }

    dfs(0)

    return result
}
```
