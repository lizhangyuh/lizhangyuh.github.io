二分查找标准模板

时间复杂度：O(logn)
空间复杂度：O(1)
<!-- more -->
```Go
func search(nums []int, target int) int {
    left, right := 0, len(nums) -1

    for left < right {
        mid := (left + right) / 2
		// 找到第一个大于等于target的数
        if nums[mid] >= target {
	        right = mid
        }else{
            left = mid + 1
        }
    }

	// 这个数等于target则返回下标
    if nums[right] == target {
        return right
    }
    
    return -1
}
```
