递归处理

```Go
func decodeString(s string) string {
    // fmt.Printf("s: %v\n", s)
    ans := ""

    l, r, lc := 0, 0, 0
    n := len(s)
    count := ""

    for i := 0; i < n; i++ {

        if s[i] == '[' {
            if lc == 0 {
                l = i + 1
            }
            lc++
            continue
        }

        if s[i] == ']' {
            lc--
            if lc == 0 {
                r = i
                temp := decodeString(s[l:r])
                // fmt.Printf("temp: %v\n", s[l:r])
                cnt, _ := strconv.Atoi(count)
                
                
                for j := 0; j < cnt; j++ {
                    ans += string(temp)
                }
                count = ""
            }

            continue
        }

        if lc == 0 {
            if s[i] < '0' || s[i] > '9' {
                ans += string(s[i])
            } else  {
                count += string(s[i])
                // fmt.Printf("count: %v\n", count)
            }
        }

    }

    // fmt.Printf("ans: %v\n", ans)
    return ans
}
```
